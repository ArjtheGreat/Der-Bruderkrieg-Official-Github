ENG_level_up_duma_decision = {
	if = {
		limit = {
			has_country_flag = ENG_sided_with_halifax_campaign
		}
		add_to_variable = {
			tooltip = ENG_increase_halifaxs_support_by_1
			halifax_support = 1
		}
	}
	if = {
		limit = {
			has_country_flag = ENG_chose_churchill
		}
		add_to_variable = {
			tooltip = ENG_increase_churchills_support_by_1
			churchill_support = 1
		}
	}
	if = {
		limit = {
			has_country_flag = ENG_sided_with_attlee_campaign
		}
		add_to_variable = {
			tooltip = ENG_increase_attlees_support_by_1
			attlee_support = 1
		}
	}
	if = {
		limit = {
			has_country_flag = ENG_sided_with_mosley_campaign
		}
		add_to_variable = {
			tooltip = ENG_increase_mosleys_support_by_1
			mosley_support = 1
		}
	}
}
