on_actions = {
    on_startup = {
        effect = {
            FRA = {
				set_variable = {FRA_emperor_approve_var = 50}
				set_variable = {FRA_senate_var = 57}
				set_variable = {FRA_chamber_of_deputies_var = 50}
				set_variable = {FRA_doctrinaire_party_support = 18}
				set_variable = {FRA_leftist_party_support = 91}
				set_variable = {FRA_business_leaders = 0}
				set_variable = {FRA_workers_support = 61}
			}
        }
	}
	on_capitulation = {
		effect = {
			if = {
				limit = {
					FROM = {
						tag = FRA
					}
					tag = FPR
				}
				FROM = {
					set_country_flag = FRA_peasentrevolt_crushed_flag
				}
			}
			
		}
	}

	on_weekly_FRA = {
		effect = {
			if = {
				limit = { 
					FRA = {OR = {NOT = {has_country_flag = FRA_peasentrevolt_crushed_flag}  has_war_with = FPR }   }
				}
				random_owned_controlled_state = {
					limit = {
						OR = { has_state_category = rural has_state_category = pastoral has_state_category = town }
						is_on_continent = europe
					}
					THIS = {
						save_global_event_target_as = france_peasant_damaged_state
					}
				}
				random_list = {
					50 = {
						country_event = francestart.9
					}
					50 = {
						country_event = francestart.10
					}
				}
			}
		}
	}
}
