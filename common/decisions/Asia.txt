		}

		fire_only_once = yes 
	
		activation = {
		  OR = { tag = GZC tag = YUN }
		  has_global_flag = south_china_war_activation
		}
	
		visible = {
		  OR = { tag = GZC tag = YUN }
		  has_global_flag = south_china_war_activation
		} 
	
		#cancel_if_not_visible = yes
	
		is_good = yes

			if = {
				limit = {

				limit = {

						var = YUN.surrender_progress

						compare = greater_than

				}
				GZC = {
					country_event = south_china_war.2

			}
